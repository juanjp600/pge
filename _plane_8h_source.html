46 class Plane {
47  public:
48  Vector3f normal;
49  float distanceFromOrigin;
50 
51  constexpr Plane() : normal(0.f, 1.f, 0.f), distanceFromOrigin(0.f) { }
52 
53  constexpr Plane(const Vector3f& p1, const Vector3f& p2, const Vector3f& p3)
54  : normal((p3 - p1).crossProduct(p2 - p1).normalize()),
55  distanceFromOrigin(normal.dotProduct(p1)) { }
56 
57  constexpr Plane(const Vector3f& norm, const Vector3f& pointOnPlane)
58  : normal(norm.normalize()),
59  distanceFromOrigin(normal.dotProduct(pointOnPlane)) { }
60 
61  constexpr Plane(const Vector3f& norm, float distOrig)
62  : normal(norm.normalize()),
63  distanceFromOrigin(distOrig) { }
64 
65  constexpr bool operator==(const Plane& other) const {
66  return normal == other.normal && distanceFromOrigin == other.distanceFromOrigin;
67  }
68 
69  constexpr bool operator!=(const Plane& other) const {
70  return normal != other.normal || distanceFromOrigin != other.distanceFromOrigin;
71  }
72 
73  constexpr bool equals(const Plane& other, float epsilon = Math::EPSILON_DEFAULT) const {
74  return normal.equals(other.normal, epsilon) && Math::equalFloats(distanceFromOrigin, other.distanceFromOrigin, epsilon);
75  }
76 
77  constexpr int onPlane(const Vector3f& co, float epsilon = Math::EPSILON_DEFAULT) const {
78  float res = normal.dotProduct(co) - distanceFromOrigin;;
79  if (Math::equalFloats(epsilon, 0.f)) { return 0; }
80  if (res < 0) { return -1; }
81  return 1;
82  }
83 
84  constexpr bool intersects(const Line3f& line, Vector3f& intersectionPoint, float& coveredAmount, bool ignoreDirection = false, bool ignoreSegment = false) const {
85  // http://softsurfer.com/Archive/algorithm_0104/algorithm_0104B.htm#Line%20Intersections
86  // http://paulbourke.net/geometry/planeline/
87 
88  Vector3f dir = line.pointB - line.pointA;
89  float denominator = -normal.dotProduct(dir);
90  float numerator = normal.dotProduct(line.pointA + normal * -distanceFromOrigin);
91  if (Math::equalFloats(denominator, 0.f) || (!ignoreDirection && denominator < 0)) { return false; }
92  float u = numerator / denominator;
93  if (!ignoreSegment && (u < 0 || u > 1)) { return false; }
94  intersectionPoint = line.pointA + dir * u;
95  coveredAmount = u;
96  return true;
97  }
98 
99  static constexpr bool intersect(const Plane& p1, const Plane& p2, const Plane& p3, Vector3f& intersectionPoint) {
100  // http://paulbourke.net/geometry/3planes/
101 
102  Vector3f c1 = p2.normal.crossProduct(p3.normal);
103  Vector3f c2 = p3.normal.crossProduct(p1.normal);
104  Vector3f c3 = p1.normal.crossProduct(p2.normal);
105 
106  float denom = p1.normal.dotProduct(c1);
107  // No intersection, planes must be parallel
108  if (Math::equalFloats(denom, 0.f)) { return false; }
109 
110  Vector3f numer = c1 * p1.distanceFromOrigin + c2 * p2.distanceFromOrigin + c3 * p3.distanceFromOrigin;
111  intersectionPoint = numer / denom;
112  return true;
113  }
114 };

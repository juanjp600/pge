67  constexpr bool equals(const Plane& other, float epsilon = Math::EPSILON_DEFAULT) const {
68  return normal.equals(other.normal, epsilon) && Math::equalFloats(distanceFromOrigin, other.distanceFromOrigin, epsilon);

71  constexpr float evalAtPoint(const Vector3f& co) const {
72  return normal.dotProduct(co) - distanceFromOrigin;
73  }

75  enum class PointRelation {
76  ON,
77  BELOW,
78  ABOVE,
79  };
80 
81  constexpr PointRelation onPlane(const Vector3f& co, float epsilon = Math::EPSILON_DEFAULT) const {
82  float res = evalAtPoint(co);
83  using enum PointRelation;
84  if (Math::equalFloats(epsilon, 0.f)) { return ON; }
85  if (res < 0) { return BELOW; }
86  return ABOVE;
87  }

89  constexpr bool intersects(const Line3f& line, Vector3f& intersectionPoint, float& coveredAmount, bool ignoreDirection = false, bool ignoreSegment = false) const {
90  // http://softsurfer.com/Archive/algorithm_0104/algorithm_0104B.htm#Line%20Intersections
91  // http://paulbourke.net/geometry/planeline/
92 
93  Vector3f dir = line.pointB - line.pointA;
94  float denominator = -normal.dotProduct(dir);
95  float numerator = normal.dotProduct(line.pointA + normal * -distanceFromOrigin);
96  if (Math::equalFloats(denominator, 0.f) || (!ignoreDirection && denominator < 0)) { return false; }
97  float u = numerator / denominator;
98  if (!ignoreSegment && (u < 0 || u > 1)) { return false; }
99  intersectionPoint = line.pointA + dir * u;
100  coveredAmount = u;
101  return true;
102  }
103 
104  static constexpr bool intersect(const Plane& p1, const Plane& p2, const Plane& p3, Vector3f& intersectionPoint) {
105  // http://paulbourke.net/geometry/3planes/
106 
107  Vector3f c1 = p2.normal.crossProduct(p3.normal);
108  Vector3f c2 = p3.normal.crossProduct(p1.normal);
109  Vector3f c3 = p1.normal.crossProduct(p2.normal);
110 
111  float denom = p1.normal.dotProduct(c1);
112  // No intersection, planes must be parallel
113  if (Math::equalFloats(denom, 0.f)) { return false; }
114 
115  Vector3f numer = c1 * p1.distanceFromOrigin + c2 * p2.distanceFromOrigin + c3 * p3.distanceFromOrigin;
116  intersectionPoint = numer / denom;
117  return true;
118  }
119 };
